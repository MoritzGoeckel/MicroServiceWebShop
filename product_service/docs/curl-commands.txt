
# Read Hello
curl -i -H "Content-Type: application/json" -X GET localhost:8080/hello

# Add new product, POST
curl -i -H "Content-Type: application/json" -X POST -d '{
        "name" : "Waschmaschine",
        "categorie" : "Ware"
    }' localhost:8080/products

# Read all Products
curl -i -H "Content-Type: application/json" -X GET localhost:8080/products

# Get one Product
curl -i -H "Content-Type: application/json" -X GET localhost:8080/products/1

# DELETE one Product
curl -i -H "Content-Type: application/json" -X DELETE localhost:8080/products/2

# Change one Product, POST
curl -i -H "Content-Type: application/json" -X POST -d '{
        "name" : "Auto",
        "categorie" : "Luxus-Ware"
    }' localhost:8080/products/2
